import pandas as pd

# Load the dataset
file_path = "avocado.csv"  # Update with your file path
avocado_data = pd.read_csv(file_path)

# Clean the data by dropping irrelevant columns
cleaned_data = avocado_data.drop(columns=["Unnamed: 0"])

# Convert the 'Date' column to datetime format for easier analysis
cleaned_data['Date'] = pd.to_datetime(cleaned_data['Date'])

import matplotlib.pyplot as plt
import seaborn as sns

# Set up the plotting style
sns.set(style="whitegrid")

# Group data by Date and type to calculate average prices
price_trends = cleaned_data.groupby(['Date', 'type'])['AveragePrice'].mean().reset_index()

# Plot the price trends
plt.figure(figsize=(12, 6))
sns.lineplot(data=price_trends, x='Date', y='AveragePrice', hue='type', marker='o')
plt.title("Average Avocado Prices Over Time by Type", fontsize=16)
plt.xlabel("Date", fontsize=12)
plt.ylabel("Average Price (USD)", fontsize=12)
plt.legend(title="Avocado Type")
plt.tight_layout()
plt.show()
